2.3 Auth Middleware

🔐 /middleware/adminAuth.js
const jwt = require('jsonwebtoken');

// Verify JWT + check admin role
function requireAdmin(requiredPermission) {
  return async (req, res, next) => {
    try {
      const token = req.headers.authorization?.replace('Bearer ', '');
      if (!token) return res.status(401).json({ error: 'No token provided' });

      const decoded = jwt.verify(token, process.env.JWT_SECRET);
      req.user = decoded;

      // Check specific permission
      if (requiredPermission && !decoded.permissions[requiredPermission]) {
        return res.status(403).json({
          error: 'Insufficient permissions',
          required: requiredPermission,
          your_role: decoded.role
        });
      }

      // Log admin action for audit trail
      await logAuditAction(decoded.sub, req.method, req.path, req.body);

      next();
    } catch (err) {
      return res.status(401).json({ error: 'Invalid or expired token' });
    }
  };
}

async function logAuditAction(userId, method, path, body) {
  if (method === 'GET') return; // Don't log reads
  await db.query(`
    INSERT INTO audit_log (user_id, action, resource, details, ip_address, created_at)
    VALUES ($1, $2, $3, $4, $5, CURRENT_TIMESTAMP)
  `, [userId, method, path, JSON.stringify(body || {}), '']);
}

router.get('/metrics', requireAdmin('dashboard'), async (req, res) => {
  const [users, therapists, sessions, revenue, conversion] = await Promise.all([
    // Total Users
    db.query(`SELECT COUNT(*) as total FROM users WHERE is_active = true`),

    // Active Therapists (had session in last 30 days)
    db.query(`
      SELECT COUNT(DISTINCT t.id) as total
      FROM therapists t
      JOIN sessions s ON s.therapist_id = t.id
      WHERE s.created_at > NOW() - INTERVAL '30 days'
        AND t.verification_status = 'approved'
    `),

    // Sessions Today
    db.query(`
      SELECT COUNT(*) as total,
        COUNT(CASE WHEN status = 'live' THEN 1 END) as live_now
      FROM sessions
      WHERE DATE(created_at) = CURRENT_DATE
    `),

    // Revenue This Month (platform 40% share)
    db.query(`
      SELECT
        COALESCE(SUM(amount * 0.40), 0) as platform_revenue,
        COALESCE(SUM(amount), 0) as gross_revenue,
        COUNT(*) as transaction_count
      FROM payments
      WHERE status = 'success'
        AND purpose IN ('session_booking','group_session','couples_session','nri_session')
        AND DATE_TRUNC('month', paid_at) = DATE_TRUNC('month', CURRENT_DATE)
    `),

    // Conversion Rate (assessment → first session)
    db.query(`
      SELECT
        COUNT(DISTINCT a.user_id) as assessed,
        COUNT(DISTINCT s.user_id) as booked
      FROM assessments a
      LEFT JOIN sessions s ON s.user_id = a.user_id
      WHERE a.created_at > NOW() - INTERVAL '30 days'
    `)
  ]);

  const convRate = conversion.rows[0].assessed > 0
    ? (conversion.rows[0].booked / conversion.rows[0].assessed * 100).toFixed(1)
    : 0;

  res.json({
    total_users: users.rows[0].total,
    active_therapists: therapists.rows[0].total,
    sessions_today: sessions.rows[0].total,
    live_sessions: sessions.rows[0].live_now,
    revenue_month: {
      platform: revenue.rows[0].platform_revenue,
      gross: revenue.rows[0].gross_revenue,
      transactions: revenue.rows[0].transaction_count
    },
    conversion_rate: convRate + '%'
  });
});

router.get('/revenue/chart', requireAdmin('dashboard'), async (req, res) => {
  const months = parseInt(req.query.months) || 6;

  const result = await db.query(`
    SELECT
      TO_CHAR(DATE_TRUNC('month', paid_at), 'Mon') as month,
      DATE_TRUNC('month', paid_at) as month_date,
      COALESCE(SUM(amount * 0.40), 0)::int as platform_revenue,
      COALESCE(SUM(amount * 0.60), 0)::int as provider_revenue,
      COUNT(*) as session_count
    FROM payments
    WHERE status = 'success'
      AND purpose IN ('session_booking','group_session','couples_session','nri_session',
                       'executive_session','sleep_session')
      AND paid_at >= DATE_TRUNC('month', CURRENT_DATE) - INTERVAL '${months} months'
    GROUP BY DATE_TRUNC('month', paid_at)
    ORDER BY month_date ASC
  `);

  // Also get subscription revenue (100% platform)
  const subs = await db.query(`
    SELECT
      TO_CHAR(DATE_TRUNC('month', paid_at), 'Mon') as month,
      COALESCE(SUM(total_amount), 0)::int as sub_revenue
    FROM payments
    WHERE status = 'success'
      AND purpose IN ('subscription','provider_subscription','mydigitalclinic')
      AND paid_at >= DATE_TRUNC('month', CURRENT_DATE) - INTERVAL '${months} months'
    GROUP BY DATE_TRUNC('month', paid_at)
    ORDER BY DATE_TRUNC('month', paid_at) ASC
  `);

  res.json({
    labels: result.rows.map(r => r.month),
    platform_revenue: result.rows.map(r => r.platform_revenue),
    provider_revenue: result.rows.map(r => r.provider_revenue),
    session_count: result.rows.map(r => r.session_count),
    subscription_revenue: subs.rows.map(r => r.sub_revenue)
  });
});

// Approve or Reject verification
router.patch('/verifications/:id', requireAdmin('verifications_approve'), async (req, res) => {
  const { id } = req.params;
  const { action, rejection_reason } = req.body; // action: 'approve' | 'reject'

  if (!['approve', 'reject'].includes(action)) {
    return res.status(400).json({ error: 'Action must be approve or reject' });
  }

  const newStatus = action === 'approve' ? 'approved' : 'rejected';

  // Update therapist record
  await db.query(`
    UPDATE therapists
    SET verification_status = $1,
        verified_by = $2,
        verified_at = CURRENT_TIMESTAMP,
        rejection_reason = $3,
        updated_at = CURRENT_TIMESTAMP
    WHERE id = $4
  `, [newStatus, req.user.sub, rejection_reason || null, id]);

  // Update Zoho Desk ticket (Blueprint transition)
  const therapist = await db.query('SELECT zoho_ticket_id, name, phone FROM therapists WHERE id=$1', [id]);
  if (therapist.rows[0]?.zoho_ticket_id) {
    await updateZohoTicket(therapist.rows[0].zoho_ticket_id, {
      status: action === 'approve' ? 'Approved' : 'Rejected',
      comment: action === 'approve'
        ? `Credentials verified by ${req.user.role}. Provider activated.`
        : `Rejected: ${rejection_reason}. Provider notified.`
    });
  }

  // Send WhatsApp notification to therapist
  await sendWhatsApp(therapist.rows[0].phone, action === 'approve'
    ? `🎉 Welcome to MANAS360! Your credentials are verified. Login to start receiving patients.`
    : `Your MANAS360 verification needs attention: ${rejection_reason}. Please resubmit.`
  );

  // Emit real-time update to all admin dashboards
  io.to('admin-room').emit('verification-updated', { id, status: newStatus });

  res.json({ success: true, id, status: newStatus });
});

const { zohoDesk } = require('../services/zohoDesk.service');

router.get('/tickets', requireAdmin('dashboard'), async (req, res) => {
  const { department, status, priority, limit } = req.query;

  // Filter by user's IAM department access
  const allowedDepts = req.user.departments.includes('ALL')
    ? null // Super admin sees everything
    : req.user.departments;

  const tickets = await zohoDesk.listTickets({
    department: department !== 'all' ? department : undefined,
    status: status || 'Open',
    priority: priority,
    limit: parseInt(limit) || 20,
    sortBy: 'createdTime',
    sortOrder: 'desc',
    // IAM filter: only show tickets from allowed departments
    ...(allowedDepts && { departmentIds: allowedDepts })
  });

  res.json({
    tickets: tickets.data.map(t => ({
      id: t.ticketNumber,
      subject: t.subject,
      status: t.status,
      priority: t.priority,
      department: t.departmentId,
      assignee: t.assignee?.name,
      contact: t.contact?.lastName,
      created: t.createdTime,
      due: t.dueDate,
      // Blueprint state
      blueprint_status: t.customFields?.cf_blueprint_state
    })),
    count: tickets.data.length
  });
});

async function updateZohoTicket(ticketId, updates) {
  const accessToken = await getZohoAccessToken();
  return axios.patch(
    `https://desk.zoho.in/api/v1/tickets/${ticketId}`,
    { status: updates.status, comment: { content: updates.comment, isPublic: false } },
    { headers: { Authorization: `Zoho-oauthtoken ${accessToken}`, orgId: process.env.ZOHO_ORG_ID } }
  );
}

// Socket.IO namespace for admin dashboard
io.of('/admin').use(async (socket, next) => {
  // Verify admin JWT on WebSocket connection
  const token = socket.handshake.auth.token;
  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    if (!decoded.permissions.dashboard) throw new Error('No dashboard access');
    socket.user = decoded;
    next();
  } catch (err) {
    next(new Error('Unauthorized'));
  }
});

io.of('/admin').on('connection', (socket) => {
  socket.join('admin-room');

  // Send current live sessions on connect
  sendLiveSessions(socket);

  // Agora webhook: session started
  agoraEvents.on('session-started', (data) => {
    io.of('/admin').to('admin-room').emit('session-started', {
      session_id: data.session_id,
      therapist: data.therapist_name,
      patient: data.patient_name, // Anonymized unless crisis
      type: data.session_type,
      started_at: data.started_at,
      agora_channel: data.channel_id
    });
  });

  // Agora webhook: session ended
  agoraEvents.on('session-ended', (data) => {
    io.of('/admin').to('admin-room').emit('session-ended', {
      session_id: data.session_id,
      duration_minutes: data.duration,
      outcome: data.outcome // completed, dropped, no_show
    });
  });

  // Crisis alert (highest priority)
  crisisEvents.on('crisis-detected', (data) => {
    io.of('/admin').to('admin-room').emit('crisis-alert', {
      patient_id: data.patient_id,
      severity: data.severity, // high, critical
      trigger: data.trigger_keywords,
      session_id: data.active_session_id,
      timestamp: new Date()
    });
  });
});

router.post('/offers/publish', requireAdmin('offers_edit'), async (req, res) => {
  // Snapshot current active offers into published cache
  const activeOffers = await db.query(`
    SELECT icon, text, badge_type, badge_text
    FROM marquee_offers
    WHERE is_active = true AND is_deleted = false
    ORDER BY sort_order ASC
  `);

  // Store in Redis for fast public access (no auth needed)
  await redis.set('marquee_offers_live', JSON.stringify(activeOffers.rows));

  // Log publish event
  await db.query(`
    INSERT INTO audit_log (user_id, action, resource, details)
    VALUES ($1, 'PUBLISH', 'marquee_offers', $2)
  `, [req.user.sub, JSON.stringify({ count: activeOffers.rowCount })]);

  res.json({ success: true, published_count: activeOffers.rowCount });
});

// PUBLIC endpoint (no auth — called by landing page)

    FROM marquee_offers
    WHERE is_active = true AND is_deleted = false
    ORDER BY sort_order ASC
  `);
  res.json({ offers: result.rows });
});

router.get('/pricing/:category', requireAdmin('dashboard'), async (req, res) => {
  const { category } = req.params; // patient, provider, corporate, sessions, addons

  const result = await db.query(`
    SELECT * FROM pricing_versions
    WHERE category = $1 AND status = 'live'
    ORDER BY created_at DESC LIMIT 1
  `, [category]);

  if (result.rows.length === 0) return res.status(404).json({ error: 'No pricing found' });

  res.json({
    version: result.rows[0].version,
    category,
    pricing: JSON.parse(result.rows[0].pricing_data),
    effective_from: result.rows[0].effective_from,
    approved_by: result.rows[0].approved_by
  });
});

// CREATE pricing draft (does NOT go live until approved)
router.post('/pricing/draft', requireAdmin('pricing_edit'), async (req, res) => {
  const { category, pricing_data, notes } = req.body;

  // Get next version number
  const maxVer = await db.query(
    'SELECT COALESCE(MAX(version),0)+1 as next FROM pricing_versions WHERE category=$1', [category]
  );

  const result = await db.query(`
    INSERT INTO pricing_versions (category, version, pricing_data, status, notes, created_by)
    VALUES ($1, $2, $3, 'draft', $4, $5)
    RETURNING *
  `, [category, maxVer.rows[0].next, JSON.stringify(pricing_data), notes, req.user.sub]);

  res.json({ success: true, draft: result.rows[0] });
});

// APPROVE draft → set as live (archives previous live version)
router.post('/pricing/draft/:id/approve', requireAdmin('pricing_edit'), async (req, res) => {
  const { id } = req.params;

  // Get the draft
  const draft = await db.query('SELECT * FROM pricing_versions WHERE id=$1 AND status=$2', [id, 'draft']);
  if (draft.rows.length === 0) return res.status(404).json({ error: 'Draft not found' });

  // Archive current live version
  await db.query(`
    UPDATE pricing_versions SET status = 'archived', archived_at = CURRENT_TIMESTAMP
    WHERE category = $1 AND status = 'live'
  `, [draft.rows[0].category]);

  // Set draft as live
  await db.query(`
    UPDATE pricing_versions
    SET status = 'live', approved_by = $1, effective_from = CURRENT_TIMESTAMP
    WHERE id = $2
  `, [req.user.sub, id]);

  // Update Redis cache for fast public access
  await redis.set(`pricing_live_${draft.rows[0].category}`, draft.rows[0].pricing_data);

  // Notify all connected admins
  io.to('admin-room').emit('pricing-updated', {
    category: draft.rows[0].category,
    version: draft.rows[0].version,
    approved_by: req.user.role
  });

  res.json({ success: true, message: `${draft.rows[0].category} pricing v${draft.rows[0].version} is now live` });
});

const ZOHO_CONFIG = {
  orgId: process.env.ZOHO_DESK_ORG_ID,
  clientId: process.env.ZOHO_CLIENT_ID,
  clientSecret: process.env.ZOHO_CLIENT_SECRET,
  refreshToken: process.env.ZOHO_REFRESH_TOKEN,
  baseUrl: 'https://desk.zoho.in/api/v1',
};

let accessToken = null;
let tokenExpiry = 0;

// Auto-refresh OAuth token
async function getAccessToken() {
  if (accessToken && Date.now() < tokenExpiry) return accessToken;

  const response = await axios.post('https://accounts.zoho.in/oauth/v2/token', null, {
    params: {
      refresh_token: ZOHO_CONFIG.refreshToken,
      client_id: ZOHO_CONFIG.clientId,
      client_secret: ZOHO_CONFIG.clientSecret,
      grant_type: 'refresh_token'
    }
  });

  accessToken = response.data.access_token;
  tokenExpiry = Date.now() + (response.data.expires_in * 1000) - 60000; // 1 min buffer
  return accessToken;
}

function headers() {
  return {
    Authorization: `Zoho-oauthtoken ${accessToken}`,
    orgId: ZOHO_CONFIG.orgId,
    'Content-Type': 'application/json'
  };
}

const zohoDesk = {
  // List tickets with filters
  async listTickets({ department, status, priority, limit, sortBy, sortOrder }) {
    await getAccessToken();
    const params = { limit: limit || 20, sortBy: sortBy || 'createdTime', sortOrder: sortOrder || 'desc' };
    if (status) params.status = status;
    if (priority) params.priority = priority;

    const url = department
      ? `${ZOHO_CONFIG.baseUrl}/departments/${department}/tickets`
      : `${ZOHO_CONFIG.baseUrl}/tickets`;

    return axios.get(url, { headers: headers(), params });
