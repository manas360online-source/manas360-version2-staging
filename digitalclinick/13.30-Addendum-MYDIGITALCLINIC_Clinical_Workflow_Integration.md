# STORY 13.30-B: MyDigitalClinic — Clinical Workflow Backend
## Functions That Make the 7 Core Modules Work

**Supersedes:** Previous integration spec (pricing-focused — WRONG)  
**Context:** Story 13.30 built the pricing/subscription shell. This story builds the **actual clinical functions** inside each module. The therapist already has patients, already has a practice. This digitizes their daily workflow.

**The 7 modules from the HTML (lines 908-917 of MYDIGITALCLINIC_REFINED.html):**

| # | HTML Module Name | What It Actually Does |
|---|---|---|
| 1 | Patient Database | Register patients, store demographics, search, CSV import |
| 2 | Session Notes & Templates | CBT/Psychodynamic/Trauma templates, write notes, auto-save |
| 3 | Scheduling & SMS Reminders | Book sessions, 24h/1h WhatsApp/SMS reminders |
| 4 | 24-Hour Data Auto Purge | DPDPA: auto-delete session notes after 24h |
| 5 | Progress Tracking (PHQ-9, GAD-7) | Administer assessments, track scores over time |
| 6 | Prescriptions & Homework | Write prescriptions, assign homework, digital sign |
| 7 | Adherence Tracking | Track homework completion, medication compliance |

**NOT in scope (v2.0):** Pricing tiers, payment gateway, billing, subscription management, feature gating.

**Reuses from MANAS360 platform:** Auth (JWT + Phone OTP), SMS/WhatsApp (Heyo), email (ZeptoMail), encryption (AES-256), Prisma, Redis, Bull queues, AWS S3.

```javascript
// services/mdc/PatientService.js

const prisma = require('../../lib/prisma');
const { encrypt, decrypt } = require('../../lib/crypto');  // REUSED from MANAS360
const { parseCSV } = require('../../lib/csv');

class PatientService {

  async registerPatient(clinicId, data) {
    // Encrypt PII before storage
    const patient = await prisma.mdcPatient.create({
      data: {
        clinicId,
        firstName: data.first_name,
        lastName: data.last_name,
        phone: encrypt(data.phone),
        email: data.email ? encrypt(data.email) : null,
        dateOfBirth: data.date_of_birth ? new Date(data.date_of_birth) : null,
        gender: data.gender,
        emergencyContactName: data.emergency_contact_name,
        emergencyContactPhone: data.emergency_contact_phone 
          ? encrypt(data.emergency_contact_phone) : null,
        primaryConcern: data.primary_concern,
        referralSource: data.referral_source,
        consentGiven: data.consent_given || false,
        consentGivenAt: data.consent_given ? new Date() : null,
        preferredLanguage: data.preferred_language || 'en',
      },
    });

    // Audit log
    await this.audit(clinicId, data._userId, 'patient_created', 'patient', patient.id);
    return patient;
  }

  async listPatients(clinicId, { search, page = 1, limit = 20 }) {
    const where = { clinicId, isActive: true };
    
    if (search) {
      where.OR = [
        { firstName: { contains: search, mode: 'insensitive' } },
        { lastName: { contains: search, mode: 'insensitive' } },
        { primaryConcern: { contains: search, mode: 'insensitive' } },
      ];
    }

    const [patients, total] = await Promise.all([
      prisma.mdcPatient.findMany({
        where,
        orderBy: { updatedAt: 'desc' },
        skip: (page - 1) * limit,
        take: limit,
        select: {
          id: true, firstName: true, lastName: true, gender: true,
          primaryConcern: true, isActive: true, createdAt: true,
          // phone/email returned decrypted only in detail view, not list
        },
      }),
      prisma.mdcPatient.count({ where }),
    ]);

    return { patients, total, page, pages: Math.ceil(total / limit) };
  }

  async getPatientDetail(clinicId, patientId) {
    const patient = await prisma.mdcPatient.findFirst({
      where: { id: patientId, clinicId },
      include: {
        sessions: { orderBy: { scheduledAt: 'desc' }, take: 10 },
        assessments: { orderBy: { administeredAt: 'desc' }, take: 10 },
        homework: { orderBy: { createdAt: 'desc' }, take: 10 },
        prescriptions: { orderBy: { createdAt: 'desc' }, take: 5 },
      },
    });
    if (!patient) return null;

    // Decrypt PII for detail view
    return {
      ...patient,
      phone: decrypt(patient.phone),
      email: patient.email ? decrypt(patient.email) : null,
      emergencyContactPhone: patient.emergencyContactPhone 
        ? decrypt(patient.emergencyContactPhone) : null,
    };
  }

  async importCSV(clinicId, userId, fileBuffer) {
    const importRecord = await prisma.mdcPatientImport.create({
      data: { clinicId, fileName: 'import.csv', totalRows: 0, status: 'processing' },
    });

    // Process async (Bull queue) — REUSED pattern from MANAS360 bulk operations
    const { mdcImportQueue } = require('../../queues');
    await mdcImportQueue.add('csv_import', {
      importId: importRecord.id, clinicId, userId,
      fileBase64: fileBuffer.toString('base64'),
    });

    return { importId: importRecord.id, status: 'processing' };
  }

  async audit(clinicId, userId, action, entityType, entityId, details = null) {
    await prisma.mdcAuditLog.create({
      data: { clinicId, userId, action, entityType, entityId, details },
    });
  }
}

module.exports = new PatientService();
```

```javascript
// services/mdc/SessionService.js

const prisma = require('../../lib/prisma');
const { addHours, addMinutes, subHours, subMinutes } = require('date-fns');

class SessionService {

  async scheduleSession(clinicId, data) {
    const session = await prisma.mdcSession.create({
      data: {
        clinicId,
        patientId: data.patient_id,
        therapistUserId: data.therapist_user_id,
        scheduledAt: new Date(data.scheduled_at),
        durationMinutes: data.duration_minutes || 50,
        sessionType: data.session_type || 'in_person',
        noteTemplate: data.note_template || 'free_form',
        status: 'scheduled',
      },
    });

    // Auto-create reminders (24h + 1h) — reuses MANAS360 reminder pattern
    const sessionTime = new Date(data.scheduled_at);

    await prisma.mdcSessionReminder.createMany({
      data: [
        {
          sessionId: session.id,
          reminderType: '24h',
          channel: 'whatsapp',   // via Heyo Phone — REUSED from Story 13.5v3.0
          scheduledFor: subHours(sessionTime, 24),
        },
        {
          sessionId: session.id,
          reminderType: '1h',
          channel: 'whatsapp',
          scheduledFor: subHours(sessionTime, 1),
        },
      ],
    });

    return session;
  }

  async listSessions(clinicId, { patientId, from, to, status, page = 1, limit = 20 }) {
    const where = { clinicId };
    if (patientId) where.patientId = patientId;
    if (status) where.status = status;
    if (from || to) {
      where.scheduledAt = {};
      if (from) where.scheduledAt.gte = new Date(from);
      if (to) where.scheduledAt.lte = new Date(to);
    }

    const sessions = await prisma.mdcSession.findMany({
      where,
      include: {
        patient: { select: { id: true, firstName: true, lastName: true } },
      },
      orderBy: { scheduledAt: 'desc' },
      skip: (page - 1) * limit,
      take: limit,
    });

    return sessions;
  }

  async completeSession(sessionId, data) {
    const clinicSettings = await prisma.mdcClinicSettings.findUnique({
      where: { clinicId: data.clinic_id },
    });

    const purgeAfter = clinicSettings?.autoPurgeEnabled
      ? addHours(new Date(), clinicSettings.autoPurgeHours || 24)
      : null;

    const session = await prisma.mdcSession.update({
      where: { id: sessionId },
      data: {
        status: 'completed',
        completedAt: new Date(),
        moodBefore: data.mood_before,
        moodAfter: data.mood_after,
        autoPurgeEnabled: clinicSettings?.autoPurgeEnabled || false,
        purgeAfter,
      },
    });

    return session;
  }

  async saveNotes(sessionId, data) {
    return prisma.mdcSession.update({
      where: { id: sessionId },
      data: {
        noteTemplate: data.template,
        sessionNotes: data.notes,          // encrypted at rest via pgcrypto
        keyThemes: data.key_themes || [],
        notesDraftStatus: 'draft',
        updatedAt: new Date(),
      },
    });
  }

  async finalizeNotes(sessionId) {
    return prisma.mdcSession.update({
      where: { id: sessionId },
      data: { notesDraftStatus: 'finalized' },
    });
  }

  /**
   * AI Post-Session Summary — REUSES Claude API from AnytimeBuddy
   */
  async generateSummary(sessionId) {
    const session = await prisma.mdcSession.findUnique({
      where: { id: sessionId },
      include: { patient: { select: { firstName: true } } },
    });

    if (!session?.sessionNotes) throw new Error('No notes to summarize');

    const ClaudeService = require('../ai/ClaudeService');  // REUSED from AnytimeBuddy
    const prompt = `You are a clinical note summarizer for a mental health professional. 
Summarize the following session notes into 3-5 key takeaways. 
Be concise and clinically relevant. Do not include patient identifying information.
Patient first name (for context only): ${session.patient.firstName}
Template used: ${session.noteTemplate}

Session notes:
${session.sessionNotes}

Provide:
1. Key themes discussed (as JSON array of strings)
2. Summary (2-3 sentences)
3. Recommended follow-up actions`;

    const summary = await ClaudeService.complete(prompt, { maxTokens: 500 });

    await prisma.mdcSession.update({
      where: { id: sessionId },
      data: { sessionSummary: summary.text },
    });

    return { summary: summary.text };
  }
}

module.exports = new SessionService();
```

```javascript
const prisma = require('../../lib/prisma');

const SEVERITY_MAP = {
  phq9: [
    { max: 4, level: 'none' },

class AssessmentService {

  async recordAssessment(clinicId, data) {
    const totalScore = Object.values(data.responses).reduce((a, b) => a + b, 0);
    const severity = this.calculateSeverity(data.assessment_type, totalScore);

    // Calculate score change vs last assessment
    const previous = await prisma.mdcAssessment.findFirst({
      where: { patientId: data.patient_id, assessmentType: data.assessment_type },
      orderBy: { administeredAt: 'desc' },
    });
    const scoreChange = previous ? totalScore - previous.totalScore : null;

    const assessment = await prisma.mdcAssessment.create({
      data: {
        clinicId,
        patientId: data.patient_id,
        sessionId: data.session_id || null,
        therapistUserId: data.therapist_user_id,
        assessmentType: data.assessment_type,
        responses: data.responses,
        totalScore,
        severity,
        scoreChange,
      },
    });

    return assessment;
  }

  async getPatientProgress(patientId, assessmentType) {
    const history = await prisma.mdcAssessment.findMany({
      where: { patientId, assessmentType },
      orderBy: { administeredAt: 'asc' },
      select: {
        id: true, totalScore: true, severity: true,
        scoreChange: true, administeredAt: true,
      },
    });

    return {
      assessmentType,
      dataPoints: history,
      latestScore: history.length > 0 ? history[history.length - 1].totalScore : null,
      latestSeverity: history.length > 0 ? history[history.length - 1].severity : null,
      trend: this.calculateTrend(history),
    };
  }

  calculateSeverity(type, score) {
    const ranges = SEVERITY_MAP[type];
    if (!ranges) return 'unknown';
    for (const range of ranges) {
      if (score <= range.max) return range.level;
    }
    return 'severe';
  }

  calculateTrend(history) {
    if (history.length < 2) return 'insufficient_data';
    const last = history[history.length - 1].totalScore;
    const prev = history[history.length - 2].totalScore;
    if (last < prev) return 'improving';
    if (last > prev) return 'worsening';
    return 'stable';
  }
}

module.exports = new AssessmentService();
```

```javascript
// services/mdc/PrescriptionService.js

const prisma = require('../../lib/prisma');
const crypto = require('crypto');
const { generatePrescriptionPDF } = require('../../lib/pdf');  // REUSED from MANAS360 Rx template
const { uploadToS3, getSignedUrl } = require('../../lib/s3');   // REUSED
const { addDays } = require('date-fns');

class PrescriptionService {

  async createPrescription(clinicId, data) {
    // Generate signature hash
    const contentToSign = JSON.stringify({
      medications: data.medications,
      clinicalNotes: data.clinical_notes,
      timestamp: new Date().toISOString(),
    });
    const signatureHash = crypto.createHash('sha256').update(contentToSign).digest('hex');

    const prescription = await prisma.mdcPrescription.create({
      data: {
        clinicId,
        patientId: data.patient_id,
        sessionId: data.session_id || null,
        therapistUserId: data.therapist_user_id,
        prescriptionType: data.prescription_type || 'medication',
        medications: data.medications,
        clinicalNotes: data.clinical_notes,
        diagnosisCodes: data.diagnosis_codes || [],
        prescriberName: data.prescriber_name,
        prescriberLicense: data.prescriber_license,
        signedAt: new Date(),
        signatureHash,
        expiresAt: addDays(new Date(), 30),
      },
    });

    // Generate PDF — REUSES MANAS360 Psychiatric Prescription Template pattern
    const patient = await prisma.mdcPatient.findUnique({ where: { id: data.patient_id } });
    const pdfBuffer = await generatePrescriptionPDF({
      prescriptionId: prescription.id,
      patient: { name: `${patient.firstName} ${patient.lastName || ''}` },
      prescriber: { name: data.prescriber_name, license: data.prescriber_license },
      medications: data.medications,
      clinicalNotes: data.clinical_notes,
      signedAt: prescription.signedAt,
      signatureHash,
    });

    // Upload to S3 — REUSES MANAS360 S3 upload pattern
    const s3Key = `mdc/${clinicId}/prescriptions/${prescription.id}.pdf`;
    await uploadToS3(s3Key, pdfBuffer, 'application/pdf');

    await prisma.mdcPrescription.update({
      where: { id: prescription.id },
      data: { prescriptionPdfUrl: s3Key },
    });

    return prescription;
  }

  async deliverToPatient(prescriptionId, channel) {
    const rx = await prisma.mdcPrescription.findUnique({
      where: { id: prescriptionId },
      include: { patient: true },
    });
    if (!rx) throw new Error('Prescription not found');

    const signedUrl = await getSignedUrl(rx.prescriptionPdfUrl, 24 * 60 * 60); // 24h expiry

    if (channel === 'whatsapp') {
      // Send via Heyo WhatsApp — REUSES Story 13.5v3.0 service window pattern
      // (send as free-form message with PDF link)
      const HeyoService = require('../notifications/HeyoService');
      await HeyoService.sendWhatsApp(rx.patient.phone, 
        `Your prescription from ${rx.prescriberName} is ready. Download: ${signedUrl}`);
    } else if (channel === 'email') {
      const EmailService = require('../notifications/EmailService');
      await EmailService.send({
        to: rx.patient.email,
        subject: `Prescription from ${rx.prescriberName} — MyDigitalClinic`,
        template: 'mdc_prescription',
        data: { downloadUrl: signedUrl, prescriberName: rx.prescriberName },
      });
    }

    await prisma.mdcPrescription.update({
      where: { id: prescriptionId },
      data: { deliveredToPatient: true, deliveredAt: new Date(), deliveryChannel: channel },
    });

    return { delivered: true, channel };
  }

  async getPrescriptionPDF(prescriptionId) {
    const rx = await prisma.mdcPrescription.findUnique({ where: { id: prescriptionId } });
    if (!rx?.prescriptionPdfUrl) throw new Error('PDF not available');
    return getSignedUrl(rx.prescriptionPdfUrl, 3600); // 1h download link
  }
}
```

### DPDPAService (Auto-Purge + Audit + Export)

```javascript
// services/mdc/DPDPAService.js

const prisma = require('../../lib/prisma');

class DPDPAService {

  /**
   * Auto-purge job — runs every 6 hours via Bull queue
   * Matches HTML: "24-Hour Data Auto Purge"
   * REUSES: Bull queue infrastructure from MANAS360
   */
  async runAutoPurge() {
    const now = new Date();

    const toPurge = await prisma.mdcSession.findMany({
      where: {
        autoPurgeEnabled: true,
        purgeAfter: { lte: now },
        notesDraftStatus: { not: 'purged' },
      },
    });

    let purgedCount = 0;
    for (const session of toPurge) {
      await prisma.mdcSession.update({
        where: { id: session.id },
        data: {
          sessionNotes: null,
          sessionSummary: null,
          keyThemes: null,
          notesDraftStatus: 'purged',
          purgedAt: now,
        },
      });

      await prisma.mdcAuditLog.create({
        data: {
          clinicId: session.clinicId,
          userId: session.therapistUserId,
          action: 'notes_auto_purged',
          entityType: 'session',
          entityId: session.id,
          details: { purgeAfterHours: 24 },
        },
      });

      purgedCount++;
    }

    return { purgedCount };
  }

  /**
   * DPDPA Right to Erasure — permanent patient data deletion
   */
  async purgePatientData(clinicId, patientId, requestedBy) {
    // Delete in correct order (FK constraints)
    await prisma.mdcHomework.deleteMany({ where: { patientId, clinicId } });
    await prisma.mdcPrescription.deleteMany({ where: { patientId, clinicId } });
    await prisma.mdcAssessment.deleteMany({ where: { patientId, clinicId } });
    await prisma.mdcSessionReminder.deleteMany({
      where: { session: { patientId, clinicId } },
    });
    await prisma.mdcSession.deleteMany({ where: { patientId, clinicId } });
    await prisma.mdcPatient.delete({ where: { id: patientId } });

    await prisma.mdcAuditLog.create({
      data: {
        clinicId, userId: requestedBy,
        action: 'patient_data_purged_dpdpa',
        entityType: 'patient', entityId: patientId,
        details: { reason: 'DPDPA right to erasure request' },
      },
    });

    return { purged: true };
  }

  /**
   * DPDPA Right to Portability — export all patient data as JSON
   */
  async exportPatientData(clinicId, patientId) {
    const patient = await prisma.mdcPatient.findFirst({
      where: { id: patientId, clinicId },
      include: {
        sessions: true,
        assessments: true,
        homework: true,
        prescriptions: { select: {
          id: true, prescriptionType: true, medications: true,
          prescriberName: true, signedAt: true, expiresAt: true,
        }},
      },
    });

    if (!patient) throw new Error('Patient not found');

    return {
      exportDate: new Date().toISOString(),
      patient: {
        name: `${patient.firstName} ${patient.lastName || ''}`,
        gender: patient.gender,
        dateOfBirth: patient.dateOfBirth,
        primaryConcern: patient.primaryConcern,
      },
      sessions: patient.sessions.map(s => ({
        date: s.scheduledAt, type: s.sessionType, status: s.status,
        notes: s.notesDraftStatus !== 'purged' ? s.sessionNotes : '[PURGED]',
        summary: s.sessionSummary,
      })),
      assessments: patient.assessments.map(a => ({
        type: a.assessmentType, date: a.administeredAt,
        score: a.totalScore, severity: a.severity,
      })),
      homework: patient.homework.map(h => ({
        assignment: h.assignmentText, dueDate: h.dueDate,
        status: h.status, feedback: h.patientFeedback,
      })),
      prescriptions: patient.prescriptions,
    };
  }

  async getAuditLog(clinicId, { page = 1, limit = 50 }) {
    const [logs, total] = await Promise.all([
      prisma.mdcAuditLog.findMany({
        where: { clinicId },
        orderBy: { createdAt: 'desc' },
        skip: (page - 1) * limit,
        take: limit,
      }),
      prisma.mdcAuditLog.count({ where: { clinicId } }),
    ]);
    return { logs, total, page, pages: Math.ceil(total / limit) };
  }
}

module.exports = new DPDPAService();
```
