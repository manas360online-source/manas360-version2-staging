CREATE TABLE notification_log (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
org_id UUID NOT NULL REFERENCES organisations(id),
type VARCHAR(50) NOT NULL, -- 'trial_d7' | 'trial_d2' | 'trial_d0'
sent_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
UNIQUE(org_id, type)
);

exports.checkAndNotify = async () => {
const now = new Date();
const results = { processed: 0, sent: 0 };

// Fetch all trial orgs not yet expired
const orgs = await db.query(
`SELECT o.id, o.name, o.trial_ends_at, o.admin_user_id,
u.email, u.name AS admin_name
FROM organisations o JOIN users u ON u.id=o.admin_user_id
WHERE o.subscription_status='trial' AND o.trial_ends_at > NOW()`);

for (const org of orgs.rows) {
results.processed++;
const days_left = Math.ceil(
(new Date(org.trial_ends_at)-now)/(1000*60*60*24));
let type = null;
if (days_left === 7) type = 'trial_d7';
if (days_left === 2) type = 'trial_d2';
if (days_left === 0) type = 'trial_d0';
if (!type) continue;

// Avoid duplicate sends
const already = await db.query(
`SELECT 1 FROM notification_log WHERE org_id=$1 AND type=$2`,
[org.id, type]);
if (already.rows.length) continue;

// Get live stats for personalised email
const stats = await getLiveStats(org.id);
await mailer.sendTrialReminder({
to: org.email, admin_name: org.admin_name,
clinic_name: org.name, days_left, stats,
payment_url: `${process.env.APP_URL}/clinic/billing?org=${org.id}`
});

await db.query(
`INSERT INTO notification_log(org_id,type) VALUES($1,$2)
ON CONFLICT DO NOTHING`, [org.id, type]);
results.sent++;
}
return results;
};

// ZeptoMail template variables:
// trial_d7: "You've logged {{sessions}} sessions and added {{patients}} patients.
// Your free trial ends in 7 days."
// trial_d2: "2 days left. Don't lose your {{sessions}} sessions of progress."
// trial_d0: "Your trial has ended. Upgrade today to keep all your patient records."
