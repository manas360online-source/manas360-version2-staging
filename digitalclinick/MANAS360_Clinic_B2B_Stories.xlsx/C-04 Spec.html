🗄️ DB Schema-- New table: therapy_sessions (extend existing if present)
CREATE TABLE IF NOT EXISTS therapy_sessions (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
org_id UUID NOT NULL REFERENCES organisations(id),
therapist_id UUID NOT NULL REFERENCES users(id),
patient_id UUID NOT NULL REFERENCES users(id),
session_date DATE NOT NULL DEFAULT CURRENT_DATE,
session_mode VARCHAR(20) DEFAULT 'offline', -- offline | walk_in | online
duration_mins INTEGER DEFAULT 45,
status VARCHAR(20) DEFAULT 'scheduled', -- scheduled | completed | cancelled
completed_at TIMESTAMP,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
CREATE INDEX idx_sessions_therapist_date ON therapy_sessions(therapist_id, session_date);
CREATE INDEX idx_sessions_org ON therapy_sessions(org_id);
CREATE INDEX idx_sessions_patient ON therapy_sessions(patient_id);

exports.getTodaySessions = async (req, res) => {
const { user_id: therapist_id, org_id } = req.user;
const today = new Date().toISOString().split('T')[0];

const result = await db.query(
`SELECT ts.id, ts.session_mode, ts.status, ts.duration_mins,
u.id AS patient_id, u.name AS patient_name,
EXISTS(
SELECT 1 FROM session_notes sn
WHERE sn.session_id = ts.id LIMIT 1
) AS has_notes
FROM therapy_sessions ts
JOIN users u ON u.id = ts.patient_id
WHERE ts.therapist_id=$1 AND ts.session_date=$2 AND ts.org_id=$3
ORDER BY ts.created_at ASC`,
[therapist_id, today, org_id]);

return res.json({ sessions: result.rows, date: today });
};

exports.completeSession = async (req, res) => {
const { session_id, duration_mins, session_mode } = req.body;
const { user_id: therapist_id } = req.user;

const result = await db.query(
`UPDATE therapy_sessions
SET status='completed', completed_at=NOW(),
duration_mins=$1, session_mode=$2
WHERE id=$3 AND therapist_id=$4 AND status!='completed'
RETURNING id, completed_at`,
[duration_mins||45, session_mode||'offline', session_id, therapist_id]);

if (!result.rows.length)
return res.status(404).json({ error: 'SESSION_NOT_FOUND_OR_ALREADY_DONE' });

return res.json({ success:true, ...result.rows[0] });
};

exports.logWalkIn = async (req, res) => {
const { user_id: therapist_id, org_id } = req.user;
const { patient_id, duration_mins } = req.body;
const session_id = require('uuid').v4();
const today = new Date().toISOString().split('T')[0];

await db.query(
`INSERT INTO therapy_sessions
(id,org_id,therapist_id,patient_id,session_date,session_mode,duration_mins,status,completed_at)
VALUES($1,$2,$3,$4,$5,'walk_in',$6,'completed',NOW())`,
[session_id, org_id, therapist_id, patient_id, today, duration_mins||30]);

return res.status(201).json({ success:true, session_id });
};
