ADD COLUMN IF NOT EXISTS plan_code VARCHAR(20), -- 'starter' | 'growth'
ADD COLUMN IF NOT EXISTS next_billing_date DATE;

CREATE TABLE billing_invoices (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
org_id UUID NOT NULL REFERENCES organisations(id),
amount INTEGER NOT NULL, -- in paise (₹2999 = 299900)
plan_code VARCHAR(20),
status VARCHAR(20) DEFAULT 'paid', -- paid | failed | refunded
phonepay_txn_id VARCHAR(100),
invoice_pdf_url TEXT,
billing_date DATE NOT NULL DEFAULT CURRENT_DATE,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
CREATE INDEX idx_invoices_org ON billing_invoices(org_id, billing_date DESC);

💻 Backend Code// controllers/clinicBilling.controller.js
const crypto = require('crypto');

const PLANS = {
starter: { amount: 299900, label: 'Starter', seat_limit: 5 },
growth: { amount: 799900, label: 'Growth', seat_limit: 15 }
};

exports.initiateBilling = async (req, res) => {
const { org_id } = req.user;
const { plan_code } = req.body;
const plan = PLANS[plan_code];
if (!plan) return res.status(400).json({ error: 'INVALID_PLAN' });

// PhonePe subscription create API
const payload = {
merchantId: process.env.PHONEPE_MERCHANT_ID,
merchantSubscriptionId: `MANAS360-${org_id}-${Date.now()}`,
merchantUserId: org_id,
authWorkflowType: 'PENNY_DROP',
amountType: 'FIXED',
amount: plan.amount,
frequency: 'MONTHLY',
recurringCount: 12,
redirectUrl: `${process.env.APP_URL}/clinic/billing/success`,
callbackUrl: `${process.env.API_URL}/clinic/billing/webhook`,
};
const base64 = Buffer.from(JSON.stringify(payload)).toString('base64');
const checksum= crypto.createHash('sha256')
.update(base64 + '/v3/recurring/subscription/create' + process.env.PHONEPE_SALT)
.digest('hex') + '###' + process.env.PHONEPE_SALT_INDEX;

const ppRes = await fetch('https://api.phonepe.com/apis/hermes/pg/v1/pay', {
method:'POST', headers:{ 'Content-Type':'application/json', 'X-VERIFY':checksum },
body: JSON.stringify({ request: base64 })
});
const ppData = await ppRes.json();
return res.json({
payment_url: ppData.data.instrumentResponse.redirectInfo.url,
subscription_id: payload.merchantSubscriptionId
});
};

exports.handleWebhook = async (req, res) => {
// 1. Verify PhonePe checksum
const incomingChecksum = req.headers['x-verify'];
// ... checksum validation ...

const { merchantSubscriptionId, code, subscriptionDetails } = req.body;
const org_id = merchantSubscriptionId.split('-')[1];

if (code === 'SUBSCRIPTION_ACTIVATED') {
await db.query(
`UPDATE organisations
SET subscription_status='active', plan_code=$1,
phonepay_subscription_id=$2,
next_billing_date=NOW()+INTERVAL '1 month'
WHERE id=$3`,
[subscriptionDetails.frequency, merchantSubscriptionId, org_id]);
}
if (code === 'SUBSCRIPTION_PAYMENT_CAPTURED') {
await db.query(
`INSERT INTO billing_invoices(org_id,amount,phonepay_txn_id,billing_date)
VALUES($1,$2,$3,CURRENT_DATE)`,
[org_id, subscriptionDetails.amount, subscriptionDetails.transactionId]);
await db.query(
`UPDATE organisations SET next_billing_date=NOW()+INTERVAL '1 month' WHERE id=$1`,
[org_id]);
}
if (['SUBSCRIPTION_CANCELLED','SUBSCRIPTION_PAYMENT_DECLINED'].includes(code)) {
await db.query(
`UPDATE organisations SET subscription_status='expired' WHERE id=$1`, [org_id]);
}
return res.json({ received: true });
};
