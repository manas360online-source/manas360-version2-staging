📋 OverviewSession Notes Module — Optional free-text notes per session. Encrypted at rest. Editable within 24h. Admin cannot read clinical notes — privacy boundary enforced at API level.
3
🗄️ DB SchemaCREATE TABLE session_notes (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
session_id UUID NOT NULL REFERENCES therapy_sessions(id) ON DELETE CASCADE,
patient_id UUID NOT NULL REFERENCES users(id),
therapist_id UUID NOT NULL REFERENCES users(id),
content TEXT NOT NULL, -- encrypted via pgcrypto
content_enc BYTEA, -- AES-256 encrypted version (production)
is_editable BOOLEAN DEFAULT true, -- false after 24h
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
CREATE INDEX idx_notes_patient ON session_notes(patient_id, created_at DESC);
CREATE INDEX idx_notes_session ON session_notes(session_id);
CREATE INDEX idx_notes_therapist ON session_notes(therapist_id);

-- Auto-lock trigger after 24 hours
CREATE OR REPLACE FUNCTION lock_old_notes() RETURNS void AS $$
UPDATE session_notes
SET is_editable = false
WHERE created_at < NOW() - INTERVAL '24 hours'
AND is_editable = true;
$$ LANGUAGE sql;
-- Run via pg_cron: SELECT cron.schedule('lock-notes', '0 * * * *', 'SELECT lock_old_notes()');

GET /api/clinic/patient/:patient_id/notes
→ 200: { notes: [{ id, content, session_date, session_mode, is_editable }] }
Note: Only assigned therapist can GET. Clinic admin endpoint deliberately excluded.

PATCH /api/clinic/notes/:note_id
Body: { content }
→ 200: { success } or 403 if is_editable=false

exports.saveNote = async (req, res) => {
const { user_id: therapist_id } = req.user;
const { session_id, content } = req.body;

if (!content?.trim())
return res.status(400).json({ error: 'EMPTY_NOTE' });

// Verify therapist owns this session
const sess = await db.query(
`SELECT patient_id FROM therapy_sessions
WHERE id=$1 AND therapist_id=$2`, [session_id, therapist_id]);
if (!sess.rows.length)
return res.status(403).json({ error: 'NOT_YOUR_SESSION' });

const note_id = require('uuid').v4();
// Production: encrypt content with AES-256 before storing
// const encrypted = encrypt(content, process.env.NOTES_ENCRYPTION_KEY);

await db.query(
`INSERT INTO session_notes(id,session_id,patient_id,therapist_id,content)
VALUES($1,$2,$3,$4,$5)`,
[note_id, session_id, sess.rows[0].patient_id, therapist_id, content]);

// Update denormalised count on patient_profiles
await db.query(
`UPDATE patient_profiles SET notes_count = notes_count + 1
WHERE user_id=$1`, [sess.rows[0].patient_id]);

return res.status(201).json({ success:true, note_id,
created_at: new Date().toISOString() });
};

exports.getPatientNotes = async (req, res) => {
const { user_id: therapist_id } = req.user;
const { patient_id } = req.params;

// Hard privacy check: therapist must be assigned to this patient
const assignment = await db.query(
`SELECT 1 FROM patient_profiles
WHERE user_id=$1 AND assigned_to=$2`, [patient_id, therapist_id]);
if (!assignment.rows.length)
return res.status(403).json({ error: 'ACCESS_DENIED' });

const notes = await db.query(
`SELECT sn.id, sn.content, sn.is_editable, sn.created_at, sn.updated_at,
ts.session_mode, ts.session_date, ts.duration_mins
FROM session_notes sn
JOIN therapy_sessions ts ON ts.id = sn.session_id
WHERE sn.patient_id=$1 AND sn.therapist_id=$2
ORDER BY sn.created_at DESC`, [patient_id, therapist_id]);

return res.json({ notes: notes.rows });
};
