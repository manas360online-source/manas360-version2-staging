🗄️ DB SchemaCREATE TABLE clinic_insights (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
org_id UUID NOT NULL REFERENCES organisations(id),
insight_text TEXT NOT NULL,
stats_json JSONB NOT NULL, -- raw numbers used to generate insight
generated_date DATE NOT NULL DEFAULT CURRENT_DATE,
UNIQUE(org_id, generated_date)
);
CREATE INDEX idx_insights_org_date ON clinic_insights(org_id, generated_date);

💻 Backend Code// controllers/clinicDashboard.controller.js
const Anthropic = require('@anthropic-ai/sdk');
const aiClient = new Anthropic();

exports.getDashboardStats = async (req, res) => {
const { org_id } = req.user;
const today = new Date().toISOString().split('T')[0];
const weekAgo = new Date(Date.now()-7*24*60*60*1000).toISOString();

// 3 fast COUNT queries in parallel
const [sessRow, therapistRow, patientRow, orgRow] = await Promise.all([
db.query(
`SELECT
COUNT(*) FILTER(WHERE created_at > $2) AS week_count,
COUNT(*) FILTER(WHERE session_mode='offline') AS offline_count,
COUNT(*) FILTER(WHERE session_mode IN('online','walk_in')) AS online_count
FROM therapy_sessions ts JOIN users u ON u.id=ts.therapist_id
WHERE u.org_id=$1 AND ts.status='completed'`,
[org_id, weekAgo]),
db.query(
`SELECT COUNT(*) AS total FROM users
WHERE org_id=$1 AND org_role='clinic_therapist'`, [org_id]),
db.query(
`SELECT COUNT(*) AS total FROM users
WHERE org_id=$1 AND org_role='clinic_patient'`, [org_id]),
db.query(
`SELECT trial_ends_at, subscription_status FROM organisations WHERE id=$1`, [org_id])
]);

const stats = {
sessions_this_week: parseInt(sessRow.rows[0].week_count),
offline_sessions: parseInt(sessRow.rows[0].offline_count),
online_sessions: parseInt(sessRow.rows[0].online_count),
therapists_active: parseInt(therapistRow.rows[0].total),
patients_total: parseInt(patientRow.rows[0].total),
};

const org = orgRow.rows[0];
const days_remaining = org.trial_ends_at
? Math.max(0, Math.ceil((new Date(org.trial_ends_at)-new Date())/(1000*60*60*24)))
: null;

// Insight: check cache, generate if needed
const cached = await db.query(
`SELECT insight_text FROM clinic_insights WHERE org_id=$1 AND generated_date=$2`,
[org_id, today]);

let insight_text, from_cache;
if (cached.rows.length) {
insight_text = cached.rows[0].insight_text;
from_cache = true;
} else {
const r = await aiClient.messages.create({
model: 'claude-sonnet-4-6', max_tokens: 100,
messages:[{ role:'user', content:
`Clinic stats this week:\n` +
`Sessions: ${stats.sessions_this_week} (${stats.offline_sessions} offline, ${stats.online_sessions} online)\n` +
`Therapists: ${stats.therapists_active} | Patients: ${stats.patients_total}\n\n` +
`Write ONE sentence summarising activity. Then ONE operational suggestion. ` +
`Plain text, no bullets, max 30 words total.`
}]
});
insight_text = r.content[0].text;
await db.query(
`INSERT INTO clinic_insights(org_id,insight_text,stats_json,generated_date)
VALUES($1,$2,$3,$4) ON CONFLICT DO NOTHING`,
[org_id, insight_text, JSON.stringify(stats), today]);
from_cache = false;
}

return res.json({
...stats,
trial_days_remaining: days_remaining,
subscription_status: org.subscription_status,
insight: { text: insight_text, generated_at: today, from_cache }
});
};
