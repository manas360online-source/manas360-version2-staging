📋 OverviewPatient Add & Profile — Admin or therapist manually adds offline patients. No patient app required. Creates a backend record with minimal fields. Assigned therapist owns the patient view.

🗄️ DB Schema-- No new tables. Uses existing users table with org fields.
-- Optional: lightweight patient_profiles for clinical metadata

CREATE TABLE patient_profiles (
user_id UUID PRIMARY KEY REFERENCES users(id) ON DELETE CASCADE,
age INTEGER,
gender VARCHAR(20),
phq9_score INTEGER, -- last recorded, optional
notes_count INTEGER DEFAULT 0, -- denormalised for fast display
added_by UUID REFERENCES users(id),
assigned_to UUID REFERENCES users(id), -- therapist user_id
is_active BOOLEAN DEFAULT true,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
CREATE INDEX idx_pp_assigned_to ON patient_profiles(assigned_to);
CREATE INDEX idx_pp_org ON patient_profiles(user_id);

exports.addPatient = async (req, res) => {
const { org_id, user_id: added_by, org_role } = req.user;
const { name, phone, age, gender, phq9_score, assigned_therapist_id } = req.body;

if (!name || !phone || !assigned_therapist_id)
return res.status(400).json({ error: 'MISSING_REQUIRED_FIELDS' });
if (!/^[6-9]\d{9}$/.test(phone))
return res.status(400).json({ error: 'INVALID_PHONE' });

// Validate assigned therapist belongs to same org
const therapist = await db.query(
`SELECT id FROM users WHERE id=$1 AND org_id=$2 AND org_role='clinic_therapist'`,
[assigned_therapist_id, org_id]);
if (!therapist.rows.length)
return res.status(400).json({ error: 'INVALID_THERAPIST' });

const client = await db.getClient();
try {
await client.query('BEGIN');
const patient_id = require('uuid').v4();

await client.query(
`INSERT INTO users(id,name,phone,org_id,org_role,is_verified,created_at)
VALUES($1,$2,$3,$4,'clinic_patient',true,NOW())`,
[patient_id, name, phone, org_id]);

await client.query(
`INSERT INTO patient_profiles(user_id,age,gender,phq9_score,added_by,assigned_to)
VALUES($1,$2,$3,$4,$5,$6)`,
[patient_id, age||null, gender||null, phq9_score||null, added_by, assigned_therapist_id]);

await client.query('COMMIT');
return res.status(201).json({
success: true,
patient: { id: patient_id, name, phone,
assigned_to: assigned_therapist_id }
});
} catch(err) {
await client.query('ROLLBACK');
if (err.code === '23505')
return res.status(409).json({ error: 'PHONE_ALREADY_EXISTS' });
return res.status(500).json({ error: 'ADD_PATIENT_FAILED' });
} finally { client.release(); }
};

exports.listPatients = async (req, res) => {
const { org_id, org_role, user_id } = req.user;
const { therapist_id, page=1, limit=20 } = req.query;
const offset = (page-1)*limit;

// Therapists see only their patients; admins can filter by therapist
const filter_therapist = org_role === 'clinic_therapist'
? user_id : (therapist_id || null);

const q = filter_therapist
? `WHERE u.org_id=$1 AND pp.assigned_to=$2 AND pp.is_active=true`
: `WHERE u.org_id=$1 AND pp.is_active=true`;
const params = filter_therapist ? [org_id, filter_therapist] : [org_id];

const result = await db.query(
`SELECT u.id,u.name,u.phone,pp.age,pp.gender,pp.phq9_score,
pp.notes_count,pp.assigned_to,pp.created_at,
COUNT(*) OVER() AS total_count
FROM users u JOIN patient_profiles pp ON pp.user_id=u.id
${q} ORDER BY pp.created_at DESC LIMIT ${limit} OFFSET ${offset}`,
params);

const total = result.rows[0]?.total_count || 0;
return res.json({ patients: result.rows, total: parseInt(total), page: parseInt(page) });
};
