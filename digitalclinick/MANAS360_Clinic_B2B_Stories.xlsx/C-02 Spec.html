📋 OverviewTherapist Invite & Sub-Account Setup — Admin invites therapists via WhatsApp link. Therapist clicks, registers with pre-filled org context, and appears in clinic roster immediately.

🗄️ DB Schema-- New table for invite tokens
CREATE TABLE clinic_invites (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
org_id UUID NOT NULL REFERENCES organisations(id),
invite_token VARCHAR(64) UNIQUE NOT NULL,
invited_by UUID NOT NULL REFERENCES users(id),
invitee_phone VARCHAR(15),
status VARCHAR(20) DEFAULT 'pending', -- pending | accepted | revoked
expires_at TIMESTAMP NOT NULL,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
CREATE INDEX idx_invite_token ON clinic_invites(invite_token);
CREATE INDEX idx_invite_org ON clinic_invites(org_id, status);

🔌 API SpecPOST /api/clinic/therapist/invite
Headers: Authorization: Bearer <admin_token>
Body: { therapist_phone }
→ 200: { success, invite_url, expires_in: 86400 }

GET /api/clinic/therapist/join?token=<invite_token>
→ 200: { valid: true, clinic_name, org_id }
→ 401: { error: 'INVITE_EXPIRED_OR_INVALID' }

POST /api/clinic/therapist/join
Body: { invite_token, name, password }
→ 201: { success, auth_token, redirect_to: '/clinic/therapist' }

DELETE /api/clinic/therapist/invite/:id → revoke invite (admin only)
GET /api/clinic/therapist/list → roster of clinic therapists

💻 Backend Code// controllers/clinicTherapist.controller.js
const crypto = require('crypto');

exports.inviteTherapist = async (req, res) => {
const { org_id, user_id: invited_by } = req.user;
const { therapist_phone } = req.body;
if (!/^[6-9]\d{9}$/.test(therapist_phone))
return res.status(400).json({ error: 'INVALID_PHONE' });

const invite_token = crypto.randomBytes(32).toString('hex');
const expires_at = new Date(Date.now() + 24*60*60*1000); // 24 hours

await db.query(
`INSERT INTO clinic_invites(org_id,invite_token,invited_by,invitee_phone,expires_at)
VALUES($1,$2,$3,$4,$5)`,
[org_id, invite_token, invited_by, therapist_phone, expires_at]);

const org = await db.query('SELECT name FROM organisations WHERE id=$1', [org_id]);
const clinic_name = org.rows[0].name;
const invite_url = `${process.env.APP_URL}/clinic/join?token=${invite_token}`;

// WhatsApp message via MSG91
await msg91.sendWhatsApp(therapist_phone,
`Hello! You've been invited to join *${clinic_name}* on MANAS360. ` +
`Click to set up your therapist account: ${invite_url} ` +
`(Link valid for 24 hours)`);

return res.status(200).json({ success: true, invite_url, expires_in: 86400 });
};

exports.joinClinic = async (req, res) => {
const { invite_token, name, password } = req.body;

const invite = await db.query(
`SELECT * FROM clinic_invites WHERE invite_token=$1 AND status='pending'
AND expires_at > NOW()`, [invite_token]);
if (!invite.rows.length)
return res.status(401).json({ error: 'INVITE_EXPIRED_OR_INVALID' });

const { org_id, invitee_phone } = invite.rows[0];
const bcrypt = require('bcrypt');
const password_hash = await bcrypt.hash(password, 12);
const user_id = require('uuid').v4();

const client = await db.getClient();
try {
await client.query('BEGIN');
await client.query(
`INSERT INTO users(id,name,phone,password_hash,org_id,org_role,is_verified,created_at)
VALUES($1,$2,$3,$4,$5,'clinic_therapist',true,NOW())`,
[user_id, name, invitee_phone, password_hash, org_id]);
await client.query(
`UPDATE clinic_invites SET status='accepted' WHERE invite_token=$1`, [invite_token]);
await client.query('COMMIT');

const auth_token = require('jsonwebtoken').sign(
{ user_id, org_id, org_role:'clinic_therapist' },
process.env.JWT_SECRET, { expiresIn:'7d' });
return res.status(201).json({
success: true, auth_token, redirect_to: '/clinic/therapist' });
} catch(err) {
await client.query('ROLLBACK');
return res.status(500).json({ error: 'JOIN_FAILED' });
} finally { client.release(); }
};
