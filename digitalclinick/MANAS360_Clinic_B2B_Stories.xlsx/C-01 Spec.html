📋 OverviewClinic Registration & Admin Setup — Enables a clinic owner to self-register, verify via OTP, and receive a JWT-authenticated admin account with a 30-day trial. Zero impact on existing individual user flows.

🗄️ DB Schema-- MIGRATION: 001_clinic_b2b_foundation.sql

CREATE TABLE organisations (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
name VARCHAR(255) NOT NULL,
city VARCHAR(100),
admin_user_id UUID,
subscription_tier VARCHAR(50) DEFAULT 'trial',
trial_ends_at TIMESTAMP,
subscription_status VARCHAR(50) DEFAULT 'trial',
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

ALTER TABLE users
ADD COLUMN IF NOT EXISTS org_id UUID REFERENCES organisations(id),
ADD COLUMN IF NOT EXISTS org_role VARCHAR(50);
-- org_role: 'clinic_admin' | 'clinic_therapist' | 'clinic_patient' | NULL (individual)

CREATE INDEX idx_users_org_id ON users(org_id);
CREATE INDEX idx_users_org_role ON users(org_role);
CREATE INDEX idx_org_status ON organisations(subscription_status);

ALTER TABLE organisations
ADD CONSTRAINT fk_org_admin FOREIGN KEY (admin_user_id) REFERENCES users(id);

🔌 API SpecPOST /api/clinic/register/initiate
Body: { clinic_name, city, admin_name, admin_phone, admin_email }
→ 200: { success, message, otp_token (JWT 10min), expires_in }
→ 409: { error: "PHONE_ALREADY_REGISTERED" }

POST /api/clinic/register/verify
Body: { otp_token, otp_code }
→ 201: { success, clinic{id,name,city,trial_ends_at}, admin{id,name,org_role}, auth_token }
→ 401: { error: "OTP_EXPIRED_OR_INVALID" | "WRONG_OTP" }

POST /api/clinic/login/initiate → sends OTP
POST /api/clinic/login/verify → returns auth_token + redirect_to: '/clinic/dashboard'

💻 Backend Code// controllers/clinic.controller.js
const { v4: uuidv4 } = require('uuid');
const jwt = require('jsonwebtoken');
const db = require('../config/db');
const msg91 = require('../services/msg91.service');
const mailer = require('../services/zepto.service');

exports.initiateRegistration = async (req, res) => {
const { clinic_name, city, admin_name, admin_phone, admin_email } = req.body;
if (!clinic_name || !city || !admin_name || !admin_phone || !admin_email)
return res.status(400).json({ success: false, error: 'MISSING_FIELDS' });
if (!/^[6-9]\d{9}$/.test(admin_phone))
return res.status(400).json({ success: false, error: 'INVALID_PHONE' });

const dup = await db.query(
"SELECT id FROM users WHERE phone=$1 AND org_role='clinic_admin'", [admin_phone]);
if (dup.rows.length > 0)
return res.status(409).json({ success:false, error:'PHONE_ALREADY_REGISTERED' });

const otp = Math.floor(100000 + Math.random() * 900000).toString();
await msg91.sendOTP(admin_phone, otp, 'clinic_registration');
const otp_token = jwt.sign(
{ clinic_name,city,admin_name,admin_phone,admin_email,otp },
process.env.JWT_OTP_SECRET, { expiresIn: '10m' });
return res.status(200).json({ success:true, otp_token, expires_in:600 });
};

exports.verifyAndCreate = async (req, res) => {
const { otp_token, otp_code } = req.body;
let payload;
try { payload = jwt.verify(otp_token, process.env.JWT_OTP_SECRET); }
catch { return res.status(401).json({ success:false, error:'OTP_EXPIRED_OR_INVALID' }); }
if (payload.otp !== otp_code)
return res.status(401).json({ success:false, error:'WRONG_OTP' });

const { clinic_name, city, admin_name, admin_phone, admin_email } = payload;
const client = await db.getClient();
try {
await client.query('BEGIN');
const org_id = uuidv4();
const trial_ends_at = new Date(Date.now() + 30*24*60*60*1000);
await client.query(
`INSERT INTO organisations(id,name,city,subscription_tier,subscription_status,trial_ends_at)
VALUES($1,$2,$3,'trial','trial',$4)`,
[org_id, clinic_name, city, trial_ends_at]);
const user_id = uuidv4();
await client.query(
`INSERT INTO users(id,name,phone,email,org_id,org_role,is_verified,created_at)
VALUES($1,$2,$3,$4,$5,'clinic_admin',true,NOW())`,
[user_id, admin_name, admin_phone, admin_email, org_id]);
await client.query(
`UPDATE organisations SET admin_user_id=$1 WHERE id=$2`, [user_id, org_id]);
await client.query('COMMIT');
const auth_token = jwt.sign(
{ user_id, org_id, org_role:'clinic_admin' },
process.env.JWT_SECRET, { expiresIn:'7d' });
mailer.sendClinicWelcome({ to:admin_email, name:admin_name, clinic_name, trial_ends_at })
.catch(console.error);
return res.status(201).json({
success:true,
clinic:{ id:org_id, name:clinic_name, city, trial_ends_at },
admin:{ id:user_id, name:admin_name, org_role:'clinic_admin' },
auth_token });
} catch(err) {
await client.query('ROLLBACK');
return res.status(500).json({ success:false, error:'REGISTRATION_FAILED' });
} finally { client.release(); }
};
