📋 OverviewAI Session Brief (Claude API) — Single-sentence patient context generated before each session using last note + session count. Cached daily. Cost ~₹1.50/brief. Graceful fallback if no prior notes.
3
🗄️ DB SchemaCREATE TABLE session_briefs (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
patient_id UUID NOT NULL REFERENCES users(id),
therapist_id UUID NOT NULL REFERENCES users(id),
brief_text TEXT NOT NULL,
generated_at DATE NOT NULL DEFAULT CURRENT_DATE,
UNIQUE(patient_id, therapist_id, generated_at) -- one brief per patient per day
);
CREATE INDEX idx_briefs_lookup ON session_briefs(patient_id, therapist_id, generated_at);
4
🔌 API SpecGET /api/clinic/session/brief/:patient_id
Headers: Authorization: Bearer <therapist_token>
→ 200: { brief: "Last session explored work anxiety...", generated_at, from_cache: true/false }
→ 200: { brief: "No prior notes — this may be a new patient or notes not yet added.", generated_at }

💻 Backend Code// services/clinicInsight.service.js — session brief
const Anthropic = require('@anthropic-ai/sdk');
const client = new Anthropic();

exports.getSessionBrief = async (patient_id, therapist_id) => {
const today = new Date().toISOString().split('T')[0];

// 1. Check cache first
const cached = await db.query(
`SELECT brief_text FROM session_briefs
WHERE patient_id=$1 AND therapist_id=$2 AND generated_at=$3`,
[patient_id, therapist_id, today]);
if (cached.rows.length)
return { brief: cached.rows[0].brief_text, from_cache: true };

// 2. Fetch context
const [patient, lastNote, sessionCount] = await Promise.all([
db.query(`SELECT name FROM users WHERE id=$1`, [patient_id]),
db.query(
`SELECT sn.content, ts.session_date, ts.session_mode
FROM session_notes sn JOIN therapy_sessions ts ON ts.id=sn.session_id
WHERE sn.patient_id=$1 AND sn.therapist_id=$2
ORDER BY sn.created_at DESC LIMIT 1`, [patient_id, therapist_id]),
db.query(
`SELECT COUNT(*) AS count FROM therapy_sessions
WHERE patient_id=$1 AND therapist_id=$2 AND status='completed'`,
[patient_id, therapist_id])
]);

const pname = patient.rows[0]?.name || 'Patient';
const note = lastNote.rows[0]?.content;
const count = parseInt(sessionCount.rows[0]?.count || 0);

// 3. Fallback if no notes
if (!note || count === 0) {
const brief = count === 0
? 'First session — no prior history on platform. Start fresh.'
: 'Sessions completed but no notes recorded yet.';
return { brief, from_cache: false };
}

// 4. Claude API call
const response = await client.messages.create({
model: 'claude-sonnet-4-6',
max_tokens: 120,
messages: [{
role: 'user',
content:
`You are a session prep assistant for a therapist. ` +
`Be concise, clinical, warm. Max 2 sentences total.\n\n` +
`Patient: ${pname} | Sessions completed: ${count}\n` +
`Last session note: "${note}"\n\n` +
`Write: (1) One-sentence status summary. ` +
`(2) One-sentence suggested focus for today. Plain text, no headers.`
}]
});

const brief_text = response.content[0].text;

// 5. Cache it
await db.query(
`INSERT INTO session_briefs(patient_id,therapist_id,brief_text,generated_at)
VALUES($1,$2,$3,$4) ON CONFLICT DO NOTHING`,
[patient_id, therapist_id, brief_text, today]);

return { brief: brief_text, from_cache: false };
};
