// Steps: 1=register(auto-done) 2=invite_therapist 3=add_patient

const STEPS = ['register', 'therapist', 'patient'];

exports.getState = async (req, res) => {
const { org_id } = req.user;
const org = await db.query(
`SELECT onboarding_step, onboarding_done FROM organisations WHERE id=$1`, [org_id]);
const step = org.rows[0].onboarding_step || 1;
return res.json({
current_step: step,
current_step_name: STEPS[step-1],
steps_done: STEPS.slice(0, step-1),
steps_remaining: STEPS.slice(step),
all_done: org.rows[0].onboarding_done
});
};

exports.completeStep = async (req, res) => {
const { org_id } = req.user;
const { step } = req.body;
const step_index = STEPS.indexOf(step.replace('skip_',''));
if (step_index < 0) return res.status(400).json({ error: 'INVALID_STEP' });

const next_step = step_index + 2; // 1-indexed, move to next
const all_done = next_step > STEPS.length;

await db.query(
`UPDATE organisations
SET onboarding_step=$1, onboarding_done=$2, updated_at=NOW()
WHERE id=$3`,
[all_done ? STEPS.length : next_step, all_done, org_id]);

return res.json({
next_step: all_done ? null : STEPS[next_step-1],
all_done,
redirect_to: all_done ? '/clinic/dashboard' : null
});
};
