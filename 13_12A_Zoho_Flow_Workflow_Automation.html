// WF-1.2 Step 3: Verify therapist license via NMC/RCI API
// Input: license_number, full_name, qualification

license = input.get("license_number").trim();
name = input.get("full_name").trim();

// Call NMC Doctor Registration API
nmcResponse = invokeurl
[
  url: "https://www.nmc.org.in/MCIRest/open/getDataFromService?service=searchDoctor"
  type: POST
  parameters: {"registrationNo": license, "doctorName": name}
];

// Parse response
if (nmcResponse.get("totalCount") == 0)
{
  // Try State Medical Council fallback
  smcResponse = invokeurl
  [
    url: zoho.flow.getVariable("SMC_API_URL") + "/verify"
    type: POST
    headers: {"Authorization": "Bearer " + zoho.flow.getVariable("SMC_API_KEY")}
    parameters: {"license": license, "name": name}
  ];
  
  if (smcResponse.get("status") != "active")
  {
    return {
      "verification_status": "flagged",
      "reason": "License not found in NMC or SMC databases",
      "requires_manual_review": true
    };
  }
}

// Check for suspended/expired
docRecord = nmcResponse.get("doctors").get(0);
if (docRecord.get("status") == "Suspended" || docRecord.get("status") == "Expired")
{
  return {
    "verification_status": "rejected",
    "reason": "License " + docRecord.get("status"),
    "requires_manual_review": false
  };
}

// Name match check (fuzzy)
registeredName = docRecord.get("doctorName").toLowerCase();
inputName = name.toLowerCase();
if (!registeredName.contains(inputName.getSuffix(" ")) && !inputName.contains(registeredName.getSuffix(" ")))
{
  return {
    "verification_status": "flagged", 
    "reason": "Name mismatch: registered='" + registeredName + "' vs input='" + inputName + "'",
    "requires_manual_review": true
  };
}

return {
  "verification_status": "verified",
  "nmc_record": docRecord,
  "verified_at": zoho.currenttime
};

PhonePe Payment Integration (Deluge)

// WF-2.1: Create PhonePe payment order
merchantId = zoho.flow.getVariable("PHONEPE_MERCHANT_ID");
saltKey = zoho.flow.getVariable("PHONEPE_SALT_KEY");
saltIndex = zoho.flow.getVariable("PHONEPE_SALT_INDEX");

transactionId = "MANS-" + zoho.currenttime.toLong() + "-" + input.get("booking_id");
amount = input.get("session_price").toNumber() * 100; // paisa

payload = Map();
payload.put("merchantId", merchantId);
payload.put("merchantTransactionId", transactionId);
payload.put("merchantUserId", input.get("patient_id"));
payload.put("amount", amount);
payload.put("redirectUrl", "https://app.mans360.co/payment/callback");
payload.put("redirectMode", "REDIRECT");
payload.put("callbackUrl", zoho.flow.getVariable("ZOHO_FLOW_WEBHOOK_PAYMENT_CALLBACK"));
payload.put("paymentInstrument", {"type": "PAY_PAGE"});

// Base64 encode
payloadB64 = zoho.encryption.base64Encode(payload.toString());

// SHA256 checksum
checksumInput = payloadB64 + "/pg/v1/pay" + saltKey;
checksum = zoho.encryption.sha256(checksumInput) + "###" + saltIndex;

// Call PhonePe
response = invokeurl
[
  url: "https://api.phonepe.com/apis/hermes/pg/v1/pay"
  type: POST
  headers: {
    "Content-Type": "application/json",
    "X-VERIFY": checksum
  }
  parameters: {"request": payloadB64}
];

return {
  "transaction_id": transactionId,
  "payment_url": response.get("data").get("instrumentResponse").get("redirectInfo").get("url"),
  "status": response.get("success")
};

WF-2.3 Patient-Therapist Matching Algorithm ✅ DONE

Trigger: Sub-flow called by WF-1.3 (Assessment completion)

Deluge: 100-Point Matching Algorithm

// WF-2.3: Patient-therapist matching (100-point scoring)
patientScore = input.get("phq9_score").toNumber();
patientLang = input.get("language_pref");
patientCity = input.get("city");
patientState = input.get("state");
patientNeed = input.get("primary_need"); // depression, anxiety, trauma, etc.

// Determine required specialization from assessment
specialization = "general";
if (patientScore >= 15) { specialization = "depression"; }
if (input.get("gad7_score") != null && input.get("gad7_score").toNumber() >= 10)
{
  specialization = "anxiety";
}
if (patientNeed == "trauma") { specialization = "trauma"; }

// Query active therapists from Supabase
headers = Map();
headers.put("apikey", zoho.flow.getVariable("SUPABASE_ANON_KEY"));
headers.put("Authorization", "Bearer " + zoho.flow.getVariable("SUPABASE_SERVICE_KEY"));

therapists = invokeurl
[
  url: zoho.flow.getVariable("SUPABASE_URL") + "/rest/v1/therapists?status=eq.active&has_availability=eq.true&select=*"
  type: GET
  headers: headers
];

// Score each therapist
matches = List();
for each t in therapists
{
  score = 0;
  
  // Specialization match (40 points)
  specs = t.get("specializations"); // comma-separated
  if (specs.contains(specialization)) { score = score + 40; }
  else if (specs.contains("general")) { score = score + 15; }
  
  // Language match (20 points)
  langs = t.get("languages");
  if (langs.contains(patientLang)) { score = score + 20; }
  
  // Location match (15 points)
  if (t.get("city") == patientCity) { score = score + 15; }
  else if (t.get("state") == patientState) { score = score + 10; }
  else { score = score + 5; } // Remote is always available
  
  // Experience (10 points)
  exp = t.get("years_experience").toNumber();
  score = score + ifnull(exp * 2, 0).min(10);
  
  // Availability (10 points)
  openSlots = t.get("open_slots_this_week").toNumber();
  score = score + ifnull(openSlots, 0).min(10);
  
  // Rating (5 points)
  rating = t.get("avg_rating").toNumber();
  score = score + ifnull(rating, 0).min(5);
  
  matches.add({"therapist_id": t.get("id"), "name": t.get("name"), "score": score});
}

// Sort by score descending, return top 10
matches.sort("score", false);
topMatches = matches.subList(0, matches.size().min(10));

return {"matches": topMatches, "total_considered": therapists.size()};

therapistId = input.get("therapist_id");

// Fetch therapist history
headers = Map();
headers.put("apikey", zoho.flow.getVariable("SUPABASE_ANON_KEY"));
headers.put("Authorization", "Bearer " + zoho.flow.getVariable("SUPABASE_SERVICE_KEY"));

profile = invokeurl
[
  url: zoho.flow.getVariable("SUPABASE_URL") + "/rest/v1/therapists?id=eq." + therapistId + "&select=*"
  type: GET
  headers: headers
].get(0);

flagged = false;
reasons = List();

// Rule 1: First payout within 7 days of joining
joinDate = profile.get("created_at").toDate();
daysSinceJoin = ((zoho.currenttime.toLong() - joinDate.toLong()) / 86400000).toNumber();
if (daysSinceJoin < 7)
{
  flagged = true;
  reasons.add("First payout within " + daysSinceJoin + " days of joining");
}

// Rule 2: Payout > ₹50,000 (high-value review)
if (payoutAmount > 50000)
{
  flagged = true;
  reasons.add("High-value payout: ₹" + payoutAmount);
}

// Rule 3: No completed sessions but wallet has funds
sessions = invokeurl
[
  url: zoho.flow.getVariable("SUPABASE_URL") + "/rest/v1/sessions?therapist_id=eq." + therapistId + "&status=eq.completed&select=id"
  type: GET
  headers: headers
];
if (sessions.size() == 0)
{
  flagged = true;
  reasons.add("No completed sessions but requesting payout");
}

// Rule 4: Multiple payout requests in 24h
recentPayouts = invokeurl
[
  url: zoho.flow.getVariable("SUPABASE_URL") + "/rest/v1/payouts?therapist_id=eq." + therapistId + "&created_at=gte." + (zoho.currenttime.subDay(1)).toString("yyyy-MM-dd'T'HH:mm:ss") + "&select=id"
  type: GET
  headers: headers
];
if (recentPayouts.size() > 1)
{
  flagged = true;
  reasons.add(recentPayouts.size() + " payout requests in last 24h");
}

return {
  "flagged": flagged,
  "reasons": reasons,
  "auto_approve": !flagged,
  "requires_manual_review": (flagged && payoutAmount > 50000)
};

  }

Step 2: Supabase POST → INSERT into 'flow_errors' table

Step 3: IF severity == "CRITICAL" or severity == "HIGH":
  → Zoho Desk: Create ticket (Department: Engineering, Priority: Urgent)
  → Twilio SMS → Tech Admin (TADM) phone
  → ZeptoMail → dev-support@mans360.co

Step 4: IF severity == "MEDIUM" or severity == "LOW":
  → Zoho Desk: Create ticket (Priority: Normal)
  → ZeptoMail → dev-support@mans360.co
